def is_pal(n: int) -> bool:
    s = str(n)
    return s == s[::-1]

def solve_one(p: int):
    cnt = 0
    while not is_pal(p):
        p += int(str(p)[::-1])
        cnt += 1
    return cnt, p

t = int(input())
for _ in range(t):
    p = int(input())
    c, ans = solve_one(p)
    print(c, ans)
